## Quick Example

```typescript
import { Engine, DefaultFlowRegistry, DefaultHandlerRegistry, MemoryStore, Result } from '@flowmonkey/core';

// Define a simple handler
const greetHandler = {
  type: 'greet',
  async execute({ input }) {
    return Result.success({ message: `Hello, ${input.name}!` });
  },
};

// Define a flow
const flow = {
  id: 'greeting',
  version: '1.0.0',
  initialStepId: 'say-hello',
  steps: {
    'say-hello': {
      id: 'say-hello',
      type: 'greet',
      input: { type: 'key', key: 'user' },
      outputKey: 'greeting',
      transitions: { onSuccess: null },
    },
  },
};

// Run it
const engine = new Engine(new MemoryStore(), handlers, flows);
const { execution } = await engine.create('greeting', { user: { name: 'World' } });
await engine.run(execution.id);
```
